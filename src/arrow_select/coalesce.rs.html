34/// Concatenate multiple [`RecordBatch`]es
35///
36/// Implements the common pattern of incrementally creating output
37/// [`RecordBatch`]es of a specific size from an input stream of
38/// [`RecordBatch`]es.
39///
40/// This is useful after operations such as [`filter`] and [`take`] that produce
41/// smaller batches, and we want to coalesce them into larger
42///
43/// [`filter`]: crate::filter::filter
44/// [`take`]: crate::take::take
45///
46/// See: <https://github.com/apache/arrow-rs/issues/6692>
47///
48/// # Example
49/// ```
50/// use arrow_array::record_batch;
51/// use arrow_select::coalesce::{BatchCoalescer};
52/// let batch1 = record_batch!(("a", Int32, [1, 2, 3])).unwrap();
53/// let batch2 = record_batch!(("a", Int32, [4, 5])).unwrap();
54///
55/// // Create a `BatchCoalescer` that will produce batches with at least 4 rows
56/// let target_batch_size = 4;
57/// let mut coalescer = BatchCoalescer::new(batch1.schema(), 4);
58///
59/// // push the batches
60/// coalescer.push_batch(batch1).unwrap();
61/// // only pushed 3 rows (not yet 4, enough to produce a batch)
62/// assert!(coalescer.next_completed_batch().is_none());
63/// coalescer.push_batch(batch2).unwrap();
64/// // now we have 5 rows, so we can produce a batch
65/// let finished = coalescer.next_completed_batch().unwrap();
66/// // 4 rows came out (target batch size is 4)
67/// let expected = record_batch!(("a", Int32, [1, 2, 3, 4])).unwrap();
68/// assert_eq!(finished, expected);
69///
70/// // Have no more input, but still have an in-progress batch
71/// assert!(coalescer.next_completed_batch().is_none());
72/// // We can finish the batch, which will produce the remaining rows
73/// coalescer.finish_buffered_batch().unwrap();
74/// let expected = record_batch!(("a", Int32, [5])).unwrap();
75/// assert_eq!(coalescer.next_completed_batch().unwrap(), expected);
76///
77/// // The coalescer is now empty
78/// assert!(coalescer.next_completed_batch().is_none());
79/// ```
80///
81/// # Background
82///
83/// Generally speaking, larger [`RecordBatch`]es are more efficient to process
84/// than smaller [`RecordBatch`]es (until the CPU cache is exceeded) because
85/// there is fixed processing overhead per batch. This coalescer builds up these
86/// larger batches incrementally.
87///
88/// ```text
89/// ┌────────────────────┐
90/// │    RecordBatch     │
91/// │   num_rows = 100   │
92/// └────────────────────┘                 ┌────────────────────┐
93///                                        │                    │
94/// ┌────────────────────┐     Coalesce    │                    │
95/// │                    │      Batches    │                    │
96/// │    RecordBatch     │                 │                    │
97/// │   num_rows = 200   │  ─ ─ ─ ─ ─ ─ ▶  │                    │
98/// │                    │                 │    RecordBatch     │
99/// │                    │                 │   num_rows = 400   │
100/// └────────────────────┘                 │                    │
101///                                        │                    │
102/// ┌────────────────────┐                 │                    │
103/// │                    │                 │                    │
104/// │    RecordBatch     │                 │                    │
105/// │   num_rows = 100   │                 └────────────────────┘
106/// │                    │
107/// └────────────────────┘
108/// ```
109///
110/// # Notes:
111///
112/// 1. Output rows are produced in the same order as the input rows
113///
114/// 2. The output is a sequence of batches, with all but the last being at exactly
115///    `target_batch_size` rows.
116///
117/// 3. Eventually this may also be able to handle other optimizations such as a
118///    combined filter/coalesce operation. See <https://github.com/apache/arrow-rs/issues/6692>
119///
120#[derive(Debug)]
121pub struct BatchCoalescer {
122    /// The input schema
123    schema: SchemaRef,
124    /// output batch size
125    batch_size: usize,
126    /// In-progress buffered batches
127    buffer: Vec<RecordBatch>,
128    /// Buffered row count. Always less than `batch_size`
129    buffered_rows: usize,
130    /// Completed batches
131    completed: VecDeque<RecordBatch>,
132}
133
134impl BatchCoalescer {
135    /// Create a new `BatchCoalescer`
136    ///
137    /// # Arguments
138    /// - `schema` - the schema of the output batches
139    /// - `batch_size` - the number of rows in each output batch.
140    ///   Typical values are `4096` or `8192` rows.
141    ///
142    pub fn new(schema: SchemaRef, batch_size: usize) -> Self {
143        Self {
144            schema,
145            batch_size,
146            buffer: vec![],
147            // We will for sure store at least one completed batch
148            completed: VecDeque::with_capacity(1),
149            buffered_rows: 0,
150        }
151    }
152
153    /// Return the schema of the output batches
154    pub fn schema(&self) -> SchemaRef {
155        Arc::clone(&self.schema)
156    }
157
158    /// Push next batch into the Coalescer
159    ///
160    /// See [`Self::next_completed_batch()`] to retrieve any completed batches.
161    pub fn push_batch(&mut self, batch: RecordBatch) -> Result<(), ArrowError> {
162        if batch.num_rows() == 0 {
163            // If the batch is empty, we don't need to do anything
164            return Ok(());
165        }
166
167        let mut batch = gc_string_view_batch(batch);
168
169        // If pushing this batch would exceed the target batch size,
170        // finish the current batch and start a new one
171        while batch.num_rows() > (self.batch_size - self.buffered_rows) {
172            let remaining_rows = self.batch_size - self.buffered_rows;
173            debug_assert!(remaining_rows > 0);
174            let head_batch = batch.slice(0, remaining_rows);
175            batch = batch.slice(remaining_rows, batch.num_rows() - remaining_rows);
176            self.buffered_rows += head_batch.num_rows();
177            self.buffer.push(head_batch);
178            self.finish_buffered_batch()?;
179        }
180        // Add the remaining rows to the buffer
181        self.buffered_rows += batch.num_rows();
182        self.buffer.push(batch);
183
184        // If we have reached the target batch size, finalize the buffered batch
185        if self.buffered_rows >= self.batch_size {
186            self.finish_buffered_batch()?;
187        }
188        Ok(())
189    }
190
191    /// Concatenates any buffered batches into a single `RecordBatch` and
192    /// clears any output buffers
193    ///
194    /// Normally this is called when the input stream is exhausted, and
195    /// we want to finalize the last batch of rows.
196    ///
197    /// See [`Self::next_completed_batch()`] for the completed batches.
198    pub fn finish_buffered_batch(&mut self) -> Result<(), ArrowError> {
199        if self.buffer.is_empty() {
200            return Ok(());
201        }
202        let batch = concat_batches(&self.schema, &self.buffer)?;
203        self.buffer.clear();
204        self.buffered_rows = 0;
205        self.completed.push_back(batch);
206        Ok(())
207    }
208
209    /// Returns true if there is any buffered data
210    pub fn is_empty(&self) -> bool {
211        self.buffer.is_empty() && self.completed.is_empty()
212    }
213
214    /// Returns true if there are any completed batches
215    pub fn has_completed_batch(&self) -> bool {
216        !self.completed.is_empty()
217    }
218
219    /// Returns the next completed batch, if any
220    pub fn next_completed_batch(&mut self) -> Option<RecordBatch> {
221        self.completed.pop_front()
222    }
223}

225/// Heuristically compact `StringViewArray`s to reduce memory usage, if needed
226///
227/// Decides when to consolidate the StringView into a new buffer to reduce
228/// memory usage and improve string locality for better performance.
229///
230/// This differs from `StringViewArray::gc` because:
231/// 1. It may not compact the array depending on a heuristic.
232/// 2. It uses a precise block size to reduce the number of buffers to track.
233///
234/// # Heuristic
235///
236/// If the average size of each view is larger than 32 bytes, we compact the array.
237///
238/// `StringViewArray` include pointers to buffer that hold the underlying data.
239/// One of the great benefits of `StringViewArray` is that many operations
240/// (e.g., `filter`) can be done without copying the underlying data.
241///
242/// However, after a while (e.g., after `FilterExec` or `HashJoinExec`) the
243/// `StringViewArray` may only refer to a small portion of the buffer,
244/// significantly increasing memory usage.
245fn gc_string_view_batch(batch: RecordBatch) -> RecordBatch {
246    let (schema, columns, num_rows) = batch.into_parts();
247    let new_columns: Vec<ArrayRef> = columns
248        .into_iter()
249        .map(|c| {
250            // Try to re-create the `StringViewArray` to prevent holding the underlying buffer too long.
251            let Some(s) = c.as_string_view_opt() else {
252                return c;
253            };
254            if s.data_buffers().is_empty() {
255                // If there are no data buffers, we can just return the array as is
256                return c;
257            }
258            let ideal_buffer_size: usize = s
259                .views()
260                .iter()
261                .map(|v| {
262                    let len = (*v as u32) as usize;
263                    if len > 12 {
264                        len
265                    } else {
266                        0
267                    }
268                })
269                .sum();
270            let actual_buffer_size = s.get_buffer_memory_size();
271            let buffers = s.data_buffers();
272
273            // Re-creating the array copies data and can be time consuming.
274            // We only do it if the array is sparse
275            if actual_buffer_size > (ideal_buffer_size * 2) {
276                if ideal_buffer_size == 0 {
277                    // If the ideal buffer size is 0, all views are inlined
278                    // so just reuse the views
279                    return Arc::new(unsafe {
280                        StringViewArray::new_unchecked(
281                            s.views().clone(),
282                            vec![],
283                            s.nulls().cloned(),
284                        )
285                    });
286                }
287                // We set the block size to `ideal_buffer_size` so that the new StringViewArray only has one buffer, which accelerate later concat_batches.
288                // See https://github.com/apache/arrow-rs/issues/6094 for more details.
289                let mut buffer: Vec<u8> = Vec::with_capacity(ideal_buffer_size);
290
291                let views: Vec<u128> = s
292                    .views()
293                    .as_ref()
294                    .iter()
295                    .cloned()
296                    .map(|v| {
297                        let mut b: ByteView = ByteView::from(v);
298
299                        if b.length > 12 {
300                            let offset = buffer.len() as u32;
301                            buffer.extend_from_slice(
302                                buffers[b.buffer_index as usize]
303                                    .get(b.offset as usize..b.offset as usize + b.length as usize)
304                                    .expect("Invalid buffer slice"),
305                            );
306                            b.offset = offset;
307                            b.buffer_index = 0; // Set buffer index to 0, as we only have one buffer
308                        }
309
310                        b.into()
311                    })
312                    .collect();
313
314                let buffers = if buffer.is_empty() {
315                    vec![]
316                } else {
317                    vec![buffer.into()]
318                };
319
320                let gc_string = unsafe {
321                    StringViewArray::new_unchecked(views.into(), buffers, s.nulls().cloned())
322                };
323
324                Arc::new(gc_string)
325            } else {
326                c
327            }
328        })
329        .collect();
330    unsafe { RecordBatch::new_unchecked(schema, new_columns, num_rows) }
331}

548    /// Test for [`BatchCoalescer`]
549    ///
550    /// Pushes the input batches to the coalescer and verifies that the resulting
551    /// batches have the expected number of rows and contents.
552    #[derive(Debug, Clone)]
553    struct Test {
554        /// Batches to feed to the coalescer.
555        input_batches: Vec<RecordBatch>,
556        /// The schema. If not provided, the first batch's schema is used.
557        schema: Option<SchemaRef>,
558        /// Expected output sizes of the resulting batches
559        expected_output_sizes: Vec<usize>,
560        /// target batch size (default to 1024)
561        target_batch_size: usize,
562    }
563
564    impl Default for Test {
565        fn default() -> Self {
566            Self {
567                input_batches: vec![],
568                schema: None,
569                expected_output_sizes: vec![],
570                target_batch_size: 1024,
571            }
572        }
573    }
574
575    impl Test {
576        fn new() -> Self {
577            Self::default()
578        }
579
580        /// Set the target batch size
581        fn with_batch_size(mut self, target_batch_size: usize) -> Self {
582            self.target_batch_size = target_batch_size;
583            self
584        }
585
586        /// Extend the input batches with `batch`
587        fn with_batch(mut self, batch: RecordBatch) -> Self {
588            self.input_batches.push(batch);
589            self
590        }
591
592        /// Extends the input batches with `batches`
593        fn with_batches(mut self, batches: impl IntoIterator<Item = RecordBatch>) -> Self {
594            self.input_batches.extend(batches);
595            self
596        }
597
598        /// Specifies the schema for the test
599        fn with_schema(mut self, schema: SchemaRef) -> Self {
600            self.schema = Some(schema);
601            self
602        }
603
604        /// Extends `sizes` to expected output sizes
605        fn with_expected_output_sizes(mut self, sizes: impl IntoIterator<Item = usize>) -> Self {
606            self.expected_output_sizes.extend(sizes);
607            self
608        }
609
610        /// Runs the test -- see documentation on [`Test`] for details
611        ///
612        /// Returns the resulting output batches
613        fn run(self) -> Vec<RecordBatch> {
614            let Self {
615                input_batches,
616                schema,
617                target_batch_size,
618                expected_output_sizes,
619            } = self;
620
621            let schema = schema.unwrap_or_else(|| input_batches[0].schema());
622
623            // create a single large input batch for output comparison
624            let single_input_batch = concat_batches(&schema, &input_batches).unwrap();
625
626            let mut coalescer = BatchCoalescer::new(Arc::clone(&schema), target_batch_size);
627
628            for batch in input_batches {
629                coalescer.push_batch(batch).unwrap();
630            }
631            coalescer.finish_buffered_batch().unwrap();
632            let mut output_batches = vec![];
633            while let Some(batch) = coalescer.next_completed_batch() {
634                output_batches.push(batch);
635            }
636
637            // make sure we got the expected number of output batches and content
638            let mut starting_idx = 0;
639            assert_eq!(expected_output_sizes.len(), output_batches.len());
640            let actual_output_sizes: Vec<usize> =
641                output_batches.iter().map(|b| b.num_rows()).collect();
642            assert_eq!(
643                expected_output_sizes, actual_output_sizes,
644                "Unexpected number of rows in output batches\n\
645                Expected\n{expected_output_sizes:#?}\nActual:{actual_output_sizes:#?}"
646            );
647            let iter = expected_output_sizes
648                .iter()
649                .zip(output_batches.iter())
650                .enumerate();
651
652            for (i, (expected_size, batch)) in iter {
653                // compare the contents of the batch after normalization (using
654                // `==` compares the underlying memory layout too)
655                let expected_batch = single_input_batch.slice(starting_idx, *expected_size);
656                let expected_batch = normalize_batch(expected_batch);
657                let batch = normalize_batch(batch.clone());
658                assert_eq!(
659                    expected_batch, batch,
660                    "Unexpected content in batch {i}:\
661                    \n\nExpected:\n{expected_batch:#?}\n\nActual:\n{batch:#?}"
662                );
663                starting_idx += *expected_size;
664            }
665            output_batches
666        }
667    }

695    /// Return a RecordBatch with a StringViewArray with num_rows by repating
696    /// values over and over.
697    fn stringview_batch_repeated<'a>(
698        num_rows: usize,
699        values: impl IntoIterator<Item = Option<&'a str>>,
700    ) -> RecordBatch {
701        let schema = Arc::new(Schema::new(vec![Field::new(
702            "c0",
703            DataType::Utf8View,
704            true,
705        )]));
706
707        // Repeat the values to a total of num_rows
708        let values: Vec<_> = values.into_iter().collect();
709        let values_iter = std::iter::repeat(values.iter())
710            .flatten()
711            .cloned()
712            .take(num_rows);
713
714        let mut builder = StringViewBuilder::with_capacity(100).with_fixed_block_size(8192);
715        for val in values_iter {
716            builder.append_option(val);
717        }
718
719        RecordBatch::try_new(Arc::clone(&schema), vec![Arc::new(builder.finish())]).unwrap()
720    }

731    /// Normalize the `RecordBatch` so that the memory layout is consistent
732    /// (e.g. StringArray is compacted).
733    fn normalize_batch(batch: RecordBatch) -> RecordBatch {
734        // Only need to normalize StringViews (as == also tests for memory layout)
735        let (schema, mut columns, row_count) = batch.into_parts();
736
737        for column in columns.iter_mut() {
738            let Some(string_view) = column.as_string_view_opt() else {
739                continue;
740            };
741
742            // Re-create the StringViewArray to ensure memory layout is
743            // consistent
744            let mut builder = StringViewBuilder::new();
745            for s in string_view.iter() {
746                builder.append_option(s);
747            }
748            // Update the column with the new StringViewArray
749            *column = Arc::new(builder.finish());
750        }
751
752        let options = RecordBatchOptions::new().with_row_count(Some(row_count));
753        RecordBatch::try_new_with_options(schema, columns, &options).unwrap()
754    }
